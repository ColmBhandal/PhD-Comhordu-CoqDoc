  (r : Distance) (t t' : Time) (i : nat) (p : reachableNet n) :
  currSince m t i n p ->
  delivered ([-[baseMode m, basePosition l], l0, r-]) t' i n p ->
  forall q : adaptNotif < t', t' < t -> ~sufficient m r -> ~failSafe m ->
